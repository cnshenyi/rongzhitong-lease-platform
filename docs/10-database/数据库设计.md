# 融智通 - 数据库设计文档

## 一、总体说明

### 1.1 数据库选型
- 主数据库：PostgreSQL 15+
- 向量数据库：pgvector（PostgreSQL插件）
- 缓存：Redis 7+
- 文件存储：MinIO / 阿里云OSS

### 1.2 设计原则
1. 所有表使用UUID作为主键
2. 所有表包含通用审计字段（created_at, updated_at, created_by, deleted_at）
3. 支持软删除（deleted_at字段）
4. 多租户隔离（tenant_id字段）
5. 敏感数据加密存储

### 1.3 命名规范
- 表名：小写 + 下划线，复数形式，如 `users`, `projects`
- 字段名：小写 + 下划线，如 `user_name`, `created_at`
- 索引名：`idx_{表名}_{字段名}`
- 外键名：`fk_{表名}_{关联表名}`

## 二、核心表结构

### 2.1 用户与权限模块

#### users 用户表
```sql
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL REFERENCES tenants(id),
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100),
    phone VARCHAR(20),
    password_hash VARCHAR(255) NOT NULL,
    real_name VARCHAR(50),
    avatar_url VARCHAR(500),
    role VARCHAR(20) NOT NULL DEFAULT 'user',
    -- role: super_admin / admin / manager / user / viewer
    department VARCHAR(100),
    status VARCHAR(20) DEFAULT 'active',
    -- status: active / inactive / locked
    last_login_at TIMESTAMP,
    login_count INTEGER DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    created_by UUID,
    deleted_at TIMESTAMP,
    
    UNIQUE(tenant_id, username),
    UNIQUE(tenant_id, email),
    UNIQUE(tenant_id, phone)
);

CREATE INDEX idx_users_tenant ON users(tenant_id);
CREATE INDEX idx_users_role ON users(role);
CREATE INDEX idx_users_status ON users(status);
```

#### tenants 租户表
```sql
CREATE TABLE tenants (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    code VARCHAR(50) UNIQUE NOT NULL,
    contact_person VARCHAR(50),
    contact_phone VARCHAR(20),
    contact_email VARCHAR(100),
    logo_url VARCHAR(500),
    plan VARCHAR(20) DEFAULT 'free',
    -- plan: free / basic / pro / enterprise
    max_users INTEGER DEFAULT 5,
    expire_at TIMESTAMP,
    status VARCHAR(20) DEFAULT 'active',
    settings JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    deleted_at TIMESTAMP
);
```

#### operation_logs 操作日志表
```sql
CREATE TABLE operation_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL,
    user_id UUID NOT NULL,
    module VARCHAR(50) NOT NULL,
    action VARCHAR(50) NOT NULL,
    target_type VARCHAR(50),
    target_id UUID,
    detail JSONB,
    ip_address VARCHAR(50),
    user_agent TEXT,
    created_at TIMESTAMP DEFAULT NOW()

CREATE INDEX idx_oplogs_tenant_time ON operation_logs(tenant_id, created_at DESC);
CREATE INDEX idx_oplogs_user ON operation_logs(user_id);
CREATE INDEX idx_oplogs_module ON operation_logs(module);
```

### 2.2 项目管理模块

#### projects 项目表
```sql
CREATE TABLE projects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL REFERENCES tenants(id),
    project_no VARCHAR(50) NOT NULL,
    project_name VARCHAR(200) NOT NULL,
    
    -- 客户信息
    customer_name VARCHAR(200) NOT NULL,
    customer_credit_code VARCHAR(18),
    customer_contact VARCHAR(50),
    customer_phone VARCHAR(20),
    
    -- 项目信息
    project_type VARCHAR(20) DEFAULT 'direct_lease',
    -- direct_lease(直租) / leaseback(售后回租)
    amount DECIMAL(18,2),
    term_months INTEGER,
    annual_rate DECIMAL(8,4),
    
    -- 状态
    status VARCHAR(20) DEFAULT 'draft',
    -- draft / in_review / approved / active / completed / cancelled
    stage VARCHAR(20) DEFAULT 'application',
    -- application / due_diligence / approval / contract / active / closing
    
    -- 负责人
    owner_id UUID REFERENCES users(id),
    
    -- 审计字段
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    created_by UUID,
    deleted_at TIMESTAMP,
    
    UNIQUE(tenant_id, project_no)
);

CREATE INDEX idx_projects_tenant ON projects(tenant_id);
CREATE INDEX idx_projects_status ON projects(status);
CREATE INDEX idx_projects_customer ON projects(customer_name);
CREATE INDEX idx_projects_owner ON projects(owner_id);
```

### 2.4 发票查重验真模块

#### invoices 发票表
```sql
CREATE TABLE invoices (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL,
    project_id UUID REFERENCES projects(id),
    
    -- 发票基本信息
    invoice_type VARCHAR(20) NOT NULL,
    -- special(专票) / normal(普票) / electronic(电子) / vehicle(机动车)
    invoice_code VARCHAR(12),
    invoice_no VARCHAR(20) NOT NULL,
    invoice_date DATE NOT NULL,
    check_code VARCHAR(20),
    
    -- 购销方信息
    buyer_name VARCHAR(200),
    buyer_tax_no VARCHAR(20),
    seller_name VARCHAR(200),
    seller_tax_no VARCHAR(20),
    
    -- 金额信息
    amount DECIMAL(18,2),
    tax DECIMAL(18,2),
    total_amount DECIMAL(18,2),
    
    -- 发票明细（JSON数组）
    items JSONB,
    
    -- 文件信息
    file_url VARCHAR(500),
    file_hash VARCHAR(64),
    
    -- OCR信息
    ocr_confidence DECIMAL(5,2),
    ocr_result JSONB,
    
    -- 验真信息
    verify_status VARCHAR(20) DEFAULT 'pending',
    -- pending / verified / invalid / cancelled / unknown
    verify_result JSONB,
    verified_at TIMESTAMP,
    
    -- 查重信息
    dedup_status VARCHAR(20) DEFAULT 'pending',
    -- pending / clean / risk_low / risk_medium / risk_high
    dedup_result JSONB,
    
    -- 指纹（用于快速查重）
    fingerprint VARCHAR(64),
    
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    created_by UUID,
    deleted_at TIMESTAMP
);

CREATE UNIQUE INDEX idx_invoices_fingerprint ON invoices(fingerprint) 
    WHERE deleted_at IS NULL;
CREATE INDEX idx_invoices_tenant ON invoices(tenant_id);
CREATE INDEX idx_invoices_project ON invoices(project_id);
CREATE INDEX idx_invoices_code_no ON invoices(invoice_code, invoice_no);
CREATE INDEX idx_invoices_seller ON invoices(seller_tax_no);
CREATE INDEX idx_invoices_dedup ON invoices(dedup_status);
```

```sql
CREATE TABLE invoice_dedup_matches (
```

### 2.8 智能问答模块

#### knowledge_documents 知识文档表
```sql
CREATE TABLE knowledge_documents (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL,
    
    doc_name VARCHAR(255) NOT NULL,
```

#### knowledge_chunks 知识分段表（含向量）
```sql
CREATE EXTENSION IF NOT EXISTS vector;

CREATE TABLE knowledge_chunks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id UUID NOT NULL REFERENCES knowledge_documents(id),
    tenant_id UUID NOT NULL,
    
    content TEXT NOT NULL,
    chunk_index INTEGER,
    
    -- 元数据
    metadata JSONB,
    -- {page: 1, section: "第一章", title: "..."}
    
    -- 向量
    embedding vector(1024),
    
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_chunks_document ON knowledge_chunks(document_id);
CREATE INDEX idx_chunks_tenant ON knowledge_chunks(tenant_id);

-- 向量索引（IVFFlat）
CREATE INDEX idx_chunks_embedding ON knowledge_chunks 
    USING ivfflat (embedding vector_cosine_ops)
    WITH (lists = 100);
```
